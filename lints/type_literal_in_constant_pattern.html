Using the Linter

Use == TypeName or TypeName _ instead of type literals in patterns.

BAD:

void f(Type x) {
  if (x case int) {
    print('int');
  }
}

GOOD:

void f(Type x) {
  if (x case == int) {
    print('int');
  }
}

BAD:

void f(Object? x) {
  if (x case int) {
    print('int');
  }
}

GOOD:

void f(Object? x) {
  if (x case int _) {
    print('int');
  }
}
